    let evaluated = false
    window.onload = ()=>{
        let score = 0
        btn = document.getElementById('check')
        btn.addEventListener('click', (e)=>{
            correct = document.querySelectorAll('.True')
            if (evaluated !== true) {
                correct.forEach(item=>{
                    if (item.checked) {
                        score++
                    }
                })
                evaluated = true
            } else {
                window.alert("Exam Already Evaluated!")
            }

            if (!document.getElementById('score')) {
                const score_div = document.createElement('DIV')
                score_div.setAttribute('id', 'score')
                score_div.innerHTML = "Total Score Is : " + score.toString()
                document.body.appendChild(score_div)
            }
        })  
    }
